311 fileMessage::fileMessage(State s,
312  bool isMainF,
313  unsigned long tId,
314  std::string filename ) :
315  Message::Message( Category::FILE ),
316  isMainFile(isMainF),
317  taskId(tId),
318  name(filename)
319 {
320  DBG_M( "fileMessage() REQ" );
321  code += (unsigned char)s;
322 
323  unsigned short fnameSize = filename.size();
324 
325  unsigned long fileSize = 0;
326 
327  unsigned short checkSum;
328 
329  std::ifstream in ( filename );
330 
331  in.ignore( std::numeric_limits<std::streamsize>::max() );
332  std::streamsize length = in.gcount();
333  in.clear(); // Since ignore will have set eof.
334  in.seekg( 0, std::ios_base::beg );
335 
336  fileSize = length;
337 
338  bufferSize = 12 + fnameSize + fileSize;
339 
340  buffer = new char [bufferSize];
341 
342  buffer[0] = code;
343 
344  unsigned char second = fnameSize & 0x7F;
345 
346  if( !isMainF )
347  second | 0x80;
348 
349  buffer[1] = second;
350  buffer[2] = 0;
351  buffer[3] = 0;
352 
353  buffer[4] = tId & 0x00FF;
354  buffer[5] = ( tId >> 8 ) & 0x00FF;
355  buffer[6] = ( tId >> 16 ) & 0x00FF;
356  buffer[7] = ( tId >> 24 ) & 0x00FF;
357 
358  buffer[8] = fileSize & 0x00FF;
359  buffer[9] = ( fileSize >> 8 ) & 0x00FF;
360  buffer[10] = ( fileSize >> 16 ) & 0x00FF;
361  buffer[11] = ( fileSize >> 24 ) & 0x00FF;
362 
363  for(unsigned short i = 0; i < fnameSize; ++i )
364  buffer[ 12 + i ] = filename[i];
365 
366  unsigned long currIndex = 12 + fnameSize;
367 
368  char c;
369 
370  boost::crc_optimal<16, 0x1021, 0xFFFF, 0, false, false> crc;
371 
372  for(unsigned long i = 0; i < fileSize; ++i )
373  {
374  in.get(c);
375  DBG(c)
376  crc.process_byte (c);
377  buffer[ currIndex + i ] = c;
378  }
379 
380  in.close();
381 
382  checkSum = crc();
383 
384  buffer[2] = checkSum & 0x00FF;
385  buffer[3] = ( checkSum >> 8 ) & 0x00FF;
386 }
